一、核心材料：瓦楞纸板 全说明
1. 为什么选择纸板？
仿生特性完美匹配：瓦楞纸板天生具备「刚柔并济」的生物组织特性 —— 既有足够的刚性做结构支撑，又有纤维与瓦楞结构带来的弹性缓冲，完美模拟人手骨骼 + 软组织的适配能力，比硬塑料、金属更贴合生命材料的 “活性”。
极致普惠零门槛：随处可见，成本几乎为零；无需 3D 打印机、CNC 机床等专业设备，一把剪刀、一瓶热熔胶就能开工，新手也能零门槛上手。
环保可持续：完全可回收、可降解，对环境零污染；不会像硅胶、水凝胶等软体材料那样老化腐烂、产生工业垃圾，长期使用无维护成本。
极强可泛化性：可随意裁切、弯折、修改，几分钟就能完成结构迭代；可自由适配任何定制化需求，不用等开模、等打印，想改就改。
2. 材料选型规范
基础选型：三层瓦楞纸板（普通快递箱纸板即可），单层厚度≥2mm，优先选用无受潮、无破损、瓦楞完整的纸板。
瓦楞方向要求：所有受力结构的瓦楞纹路，必须平行于结构长度方向，最大化提升抗弯强度。
进阶可选：5 层瓦楞纸板（用于舵机安装位、弹簧连接点等核心受力区加厚）、2mm 硬纸板（用于手指支板）。
3. 核心加工工艺规范
半刀划刻折弯：所有折弯线必须用美工刀划半刀（仅划穿表层 + 瓦楞层，保留底层纸面不划穿），再沿折痕对折，保证折角垂直、无开裂、无回弹。
受力区加厚：舵机安装位、弹簧连接点、对接孔位等受力区域，必须用双层纸板加厚（总厚度≥4mm），避免长期受力变形、撕裂。
边缘处理：裁切后边缘用砂纸打磨平整，可贴透明胶带做耐磨防护，避免划断钢丝、线材。

from adafruit_pca9685 import PCA9685
from board import I2C
import time
import RPi.GPIO as GPIO

# ===================== 硬件初始化 =====================
# 初始化I2C与PCA9685舵机驱动板
i2c = I2C()
pca = PCA9685(i2c)
pca.frequency = 50  # 舵机标准PWM频率

# 360度舵机PWM参数（根据你的舵机校准）
STOP_PWM = 300    # 舵机停止
WIND_PWM = 500    # 顺时针收线（手指弯曲）
UNWIND_PWM = 100  # 逆时针放线（手指伸直）

# 手指与PCA9685通道映射（可自由修改）
FINGER_MAP = {
    "thumb": 0,     # 拇指接通道0
    "index": 1,     # 食指接通道1
    "middle": 2,    # 中指接通道2
    "ring": 3,      # 无名指接通道3
    "pinky": 4      # 小指接通道4
}

# 限位开关GPIO映射
LIMIT_SWITCH_MAP = {
    "index_flex": 17,  # 食指弯曲限位接GPIO17
    "index_extend": 27 # 食指伸直限位接GPIO27
}

# GPIO初始化
GPIO.setmode(GPIO.BCM)
for pin in LIMIT_SWITCH_MAP.values():
    GPIO.setup(pin, GPIO.IN, pull_up_down=GPIO.PUD_UP)

# ===================== 核心控制函数 =====================
def set_servo(channel, pwm_value):
    """设置单个舵机的PWM值"""
    pca.channels[channel].duty_cycle = pwm_value

def finger_flex(finger_name, flex_time=1.5):
    """单根手指弯曲动作"""
    channel = FINGER_MAP[finger_name]
    set_servo(channel, WIND_PWM)
    time.sleep(flex_time)
    set_servo(channel, STOP_PWM)
    print(f"{finger_name} 手指弯曲完成")

def finger_extend(finger_name, extend_time=1.0):
    """单根手指伸直动作（弹簧复位辅助）"""
    channel = FINGER_MAP[finger_name]
    set_servo(channel, UNWIND_PWM)
    time.sleep(extend_time)
    set_servo(channel, STOP_PWM)
    print(f"{finger_name} 手指伸直完成")

def full_grasp(grasp_time=2.0):
    """全手抓握动作（范式化基础动作）"""
    for finger in FINGER_MAP.keys():
        set_servo(FINGER_MAP[finger], WIND_PWM)
    time.sleep(grasp_time)
    for finger in FINGER_MAP.keys():
        set_servo(FINGER_MAP[finger], STOP_PWM)
    print("全手抓握动作完成")

def full_release(release_time=1.5):
    """全手松开动作（范式化基础动作）"""
    for finger in FINGER_MAP.keys():
        set_servo(FINGER_MAP[finger], UNWIND_PWM)
    time.sleep(release_time)
    for finger in FINGER_MAP.keys():
        set_servo(FINGER_MAP[finger], STOP_PWM)
    print("全手松开动作完成")

# ===================== 安全停止函数 =====================
def emergency_stop():
    """紧急停止，所有舵机断电，GPIO清理"""
    for channel in FINGER_MAP.values():
        set_servo(channel, STOP_PWM)
    pca.deinit()
    GPIO.cleanup()
    print("紧急停止完成，所有设备已安全关闭")

# ===================== 测试运行 =====================
if __name__ == "__main__":

本部分完全承接项目「普惠零门槛」的核心愿景，用全球普及率最高的树莓派作为核心控制器，全链路采用开源 Python 代码实现所有控制功能，无需专业工控设备、无需复杂的实时系统、无需焊接加工，哪怕是零基础、从没写过代码的新手，跟着步骤复制粘贴，就能实现仿生手的全功能控制、远程指令交互，甚至后续扩展具身智能能力。

2. 电源系统连接规范（强制安全标准）
plaintext
12V锂电池正极 → 船型总开关 → 分3路并联输出：
  ↳ 第一路：12V转6V降压模块输入 → 模块输出6V → PCA9685的V+引脚 + 所有舵机红线
  ↳ 第二路：12V转5V降压模块输入 → 模块输出5V → 树莓派Type-C供电口 + 传感器/继电器VCC
  ↳ 第三路（预留爬墙模块）：继电器模块COM端 → 继电器NO端 → 电磁阀/真空泵12V正极
12V锂电池负极 → 所有模块GND共地连接（树莓派、PCA9685、降压模块、继电器、传感器）
⚠️ 强制安全要求：
所有正负极接线必须用冷压端子压接，绝缘胶带包裹，严禁裸露铜线，防止短路。
所有降压模块必须先接好负载，再接输入电源，严禁空载通电。
首次通电前，必须用万用表核对正负极，确认无接反、无短路，再通电测试。

from adafruit_pca9685 import PCA9685
from board import I2C
import time
import RPi.GPIO as GPIO

# ===================== 硬件初始化（与手指模块1:1匹配） =====================
# 初始化I2C总线与PCA9685舵机驱动板
i2c = I2C()
pca = PCA9685(i2c)
pca.frequency = 50  # 舵机标准PWM频率，不可随意修改

# 360度舵机PWM参数（可根据你的舵机实际情况校准）
STOP_PWM = 300    # 舵机停止转动
WIND_PWM = 500    # 顺时针收线（手指弯曲）
UNWIND_PWM = 100  # 逆时针放线（手指伸直）

# 五手指与PCA9685通道映射（可自由修改对应关系）
FINGER_CHANNEL_MAP = {
    "thumb": 0,     # 拇指  对应通道0
    "index": 1,     # 食指  对应通道1
    "middle": 2,    # 中指  对应通道2
    "ring": 3,      # 无名指对应通道3
    "pinky": 4      # 小指  对应通道4
}

# 限位开关GPIO映射（与第二部分传感模块匹配）
LIMIT_SWITCH_MAP = {
    "index_flex": 17,  # 食指弯曲极限限位 接GPIO17
    "index_extend": 27 # 食指伸直极限限位 接GPIO27
}

# GPIO口初始化
GPIO.setmode(GPIO.BCM)
for pin in LIMIT_SWITCH_MAP.values():
    GPIO.setup(pin, GPIO.IN, pull_up_down=GPIO.PUD_UP)

# ===================== 核心控制函数（范式化封装，新手直接调用） =====================
def set_servo_single(channel, pwm_value):
    """设置单个舵机的PWM输出，底层驱动函数"""
    pca.channels[channel].duty_cycle = pwm_value

def finger_bend(finger_name, bend_time=1.5):
    """单根手指弯曲动作，可自定义弯曲时间调整弯曲幅度"""
    channel = FINGER_CHANNEL_MAP[finger_name]
    set_servo_single(channel, WIND_PWM)
    time.sleep(bend_time)
    set_servo_single(channel, STOP_PWM)
    print(f"【动作完成】{finger_name} 手指弯曲")

def finger_straight(finger_name, straight_time=1.0):
    """单根手指伸直动作，弹簧复位辅助，可自定义时间"""
    channel = FINGER_CHANNEL_MAP[finger_name]
    set_servo_single(channel, UNWIND_PWM)
    time.sleep(straight_time)
    set_servo_single(channel, STOP_PWM)
    print(f"【动作完成】{finger_name} 手指伸直")

def full_grasp(grasp_time=2.0):
    """全手抓握动作，范式化基础动作，可自定义抓握力度/时间"""
    for finger in FINGER_CHANNEL_MAP.keys():
        set_servo_single(FINGER_CHANNEL_MAP[finger], WIND_PWM)
    time.sleep(grasp_time)
    for finger in FINGER_CHANNEL_MAP.keys():
        set_servo_single(FINGER_CHANNEL_MAP[finger], STOP_PWM)
    print("【动作完成】全手抓握")

def full_release(release_time=1.5):
    """全手松开动作，范式化基础动作"""
    for finger in FINGER_CHANNEL_MAP.keys():
        set_servo_single(FINGER_CHANNEL_MAP[finger], UNWIND_PWM)
    time.sleep(release_time)
    for finger in FINGER_CHANNEL_MAP.keys():
        set_servo_single(FINGER_CHANNEL_MAP[finger], STOP_PWM)
    print("【动作完成】全手松开")

def emergency_stop():
    """紧急停止函数，所有舵机立即停止，清理GPIO，防止失控烧毁设备"""
    for channel in FINGER_CHANNEL_MAP.values():
        set_servo_single(channel, STOP_PWM)
    pca.deinit()
    GPIO.cleanup()
    print("【紧急停止】所有设备已安全关闭，程序终止")

# ===================== 测试运行（新手直接执行即可验证） =====================
if __name__ == "__main__":
    try:
        print("===== 平民化仿生手控制程序启动 =====")
        # 单根手指动作测试
        finger_bend("index")
        time.sleep(1)
        finger_straight("index")
        time.sleep(1)
        # 全手抓握测试
        full_grasp()
        time.sleep(2)
        full_release()
        print("===== 所有动作测试完成，程序正常 =====")
    except KeyboardInterrupt:
        # 按下Ctrl+C触发紧急停止
        emergency_stop()
    except Exception as e:
        # 程序出错时触发紧急停止
        print(f"【程序异常】错误信息：{e}")
        emergency_stop()
